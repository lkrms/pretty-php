# Formatter

To generate a summary of rule operations, run the following Bash command in the
relevant directory:

```bash
find . -type f -name '*.php' -print0 |
  xargs -0 grep -EoH \
    -e '->Whitespace(Before|After|Mask(Prev|Next)) +[^ =]*= +.*' \
    -e '->(addWhitespaceBefore|applyInnerMask)\([^)]*\)' |
  tr -d ' ' |
  sed -E '
s/\.php:->/ /
s/WhitespaceType:://g
s/Whitespace//g
s/\|=?/+/g
s/(&=?|~)+/-/g
s/[;)]+$//
/addBefore/ {
  s/addBefore\(/Before+/; p
  s/Before/MaskPrev/; p
  s/MaskPrev/MaskNext/
}
/applyInnerMask/ {
  s/applyInnerMask\(-?/MaskPrev-/; p
  s/MaskPrev/MaskNext/
}' |
  LC_ALL=C sort -u |
  awk -F'[ =+-]' '
              { o = $0 }
$1 $2 != l p  { op(); n = 0; p = $2 }
$1 != l       { if (l) print ""; l = $1; sub(/.*\//, "", $1); print "`" $1 "`:" }
              { sub(/^[^ =+-]+[ =+-][^ =+-]+/, "", o); a[n++] = o }
END           { op() }

function op(_c) {
  if (!p) return
  _c = length(a)
  printf "- `%s`%s", p, (_c > 1 ? "(" : "")
  for (i = 0; i < _c; i++) {
    printf "%s%s", (i > 0 ? "|" : ""), a[i]
    delete a[i]
  }
  print (_c > 1 ? ")" : "")
}'
```
